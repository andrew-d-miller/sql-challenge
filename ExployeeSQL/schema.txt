# Physical Schema 

departments
--
dept_no VARCHAR PK
dept_name VARCHAR

department_employees
--
emp_no INTEGER FK >- employees.emp_no 
dept_no VARCHAR FK >- departments.dept_no

department_manager
--
dept_no VARCHAR FK >- departments.dept_no
emp_no INTEGER FK >- employees.emp_no

employees
--
emp_no INTEGER PK
emp_title_id VARCHAR 
birth_date DATE
first_name VARCHAR
last_name VARCHAR
sex VARCHAR
hire_date DATE

salaries
--
emp_no INTEGER FK >- employees.emp_no
salary INTEGER 

titles
--
emp_no INTEGER FK >- employees.emp_no
title_id VARCHAR 
title VARCHAR
